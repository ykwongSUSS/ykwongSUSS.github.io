curves

0.1 Packages

using Distributions
using Plots 
using LaTeXStrings

These curves are based on direct functions. The x domain is usually defined by a range sequence, and the functions are applied on the range via a dot function.

1 2D Plots

1.1 Simple function plot

# note1: x is a range type
# note2: y is passed as a vector to plot

x = range(-2,2,length=50)
y = x.^3
plot(x,y, framestyle=:origin, leg=false, 
     size=(300,250),lw=2,linecolor=:black)

1.2 Polynomial functions

# using LaTeXStrings
# f, g are functions passed to plot
# hence don't need @.

x = range(-2,2, length=100)
f(x) = x^2 -5*x +2
g(x) = x^4+3

plot(x,[f,g],size=(400,300), 
  label=[L"f=x^2-5x+2" L"g=x^4+3"], lw=2)
# f,g are functions passed to plot
# annotations uses LaTeXStrings

x = range(-2,2, length=100)
f(x)= x^4 + 2
g(x)= -2*x^2+10
plot(x,[f,g], size=(400,300), leg=false,lw=3)
annotate!(-1,17,L"f(x)=x^4+3")
annotate!(0,11,L"g(x)=-2x^2+10")

1.3 Trigonometrical functions

x = range(0, 10, length=100)
y1 = sin.(x)
y2 = cos.(x)
plot(x, [y1,y2], label=["sin(x)" "cos(x)"], linewidth=2)
plot!(size=(400,250))

1.4 Exponential (Normal) Plot

x = range(-3.5, 3.5, length=100)
y3=@. (1/sqrt(2*pi))*exp(-(x^2)/2)
plot(x, y3, label=L"mu=0,σ=1")
plot!(size=(400,250))

y4=@. (1/(0.5*sqrt(2*pi)))*exp(-(x^2)/(2*0.5^2))
plot!(x,y4,label=L"mu=0,σ=0.5")

1.5 (Normal) Distribution Plot

using Distributions

plot(Normal(0,1), size=(400,250), lw=1, leg=false)
plot!(Normal(0,0.5))
